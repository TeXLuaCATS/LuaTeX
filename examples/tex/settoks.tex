\def\mymacro{My Macro}
\directlua{
  tex.settoks(0, 'Toks at 0: \string\\mymacro')
  print(tex.gettoks(0))
  print(tex.toks[0])

  tex.scantoks(1, 0, 'Toks at 1: \string\\mymacro')
  print(tex.gettoks(1))
  print(tex.toks[1])
  tex.forcehmode(true)

  tex.runtoks(0, true, true)
  tex.runtoks(1, true, true)
}
\bye
